# rawdoc Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single Go binary that fetches web pages and converts them to clean markdown for AI coding agents, with three-tier fetch strategy (plain HTTP → TLS spoofing → headless Chrome) and crawl support.

**Architecture:** Single `main` package with six files: CLI entry (main.go), three-tier HTTP client (fetch.go), BFS crawl engine (crawl.go), noise stripping + readability extraction (extract.go), DOM→Markdown converter (convert.go), and site-specific CSS selectors (sites.go). Auto-escalates through fetch tiers on failure. Crawl mode uses BFS with concurrency pool and rate limiting.

**Tech Stack:** Go 1.26, goquery (DOM), go-readability (content extraction), utls (TLS fingerprinting), go-rod (headless Chrome), stdlib flag (CLI)

---

## File Structure

| File | Responsibility |
|---|---|
| `main.go` | CLI flag parsing, mode dispatch (single vs crawl), exit codes |
| `fetch.go` | Three-tier HTTP client: Tier1 (plain), Tier2 (utls), Tier3 (go-rod). Retry/backoff. Returns raw HTML |
| `extract.go` | Noise stripping (remove script/style/nav/ads), content extraction (site-specific → semantic → readability → body fallback) |
| `convert.go` | Clean HTML DOM → Markdown string. Handles all HTML elements → MD equivalents |
| `sites.go` | Map of domain → CSS selectors for known doc frameworks |
| `crawl.go` | BFS queue, URL normalization, dedup, concurrency pool, rate limiter, output directory |
| `fetch_test.go` | Tier escalation, headers, retry logic |
| `extract_test.go` | Noise stripping, content extraction with HTML fixtures |
| `convert_test.go` | HTML → Markdown for all element types |
| `sites_test.go` | Site-specific selector matching |
| `crawl_test.go` | BFS, dedup, include/exclude, max-pages |

```go
package main

import (
	"fmt"
	"strings"

	"github.com/PuerkitoBio/goquery"
	"golang.org/x/net/html"
)

// convertToMarkdown converts an HTML selection to clean markdown.
func convertToMarkdown(sel *goquery.Selection) string {
	var buf strings.Builder
	convertNodes(&buf, sel, 0, false)
	return cleanMarkdown(buf.String())
}

func convertNodes(buf *strings.Builder, sel *goquery.Selection, depth int, inBlockquote bool) {
	sel.Contents().Each(func(i int, s *goquery.Selection) {
		node := s.Get(0)
		if node.Type == html.TextNode {
			text := node.Data
			// Collapse whitespace in non-pre contexts
			text = collapseWhitespace(text)
			if text != "" {
				buf.WriteString(text)
			}
			return
		}
		if node.Type != html.ElementNode {
			return
		}

		tag := goquery.NodeName(s)
		switch tag {
		case "h1", "h2", "h3", "h4", "h5", "h6":
			level := int(tag[1] - '0')
			prefix := strings.Repeat("#", level)
			text := strings.TrimSpace(s.Text())
			buf.WriteString("\n\n" + prefix + " " + text + "\n\n")

		case "p":
			buf.WriteString("\n\n")
			convertNodes(buf, s, depth, inBlockquote)
			buf.WriteString("\n\n")

		case "br":
			buf.WriteString("\n")

		case "hr":
			buf.WriteString("\n\n---\n\n")

		case "strong", "b":
			buf.WriteString("**")
			convertNodes(buf, s, depth, inBlockquote)
			buf.WriteString("**")

		case "em", "i":
			buf.WriteString("*")
			convertNodes(buf, s, depth, inBlockquote)
			buf.WriteString("*")

		case "code":
			// Check if parent is <pre> — handled by pre case
			if s.Parent().Is("pre") {
				return
			}
			buf.WriteString("`")
			buf.WriteString(strings.TrimSpace(s.Text()))
			buf.WriteString("`")

		case "pre":
			lang := ""
			codeEl := s.Find("code")
			var codeText string
			if codeEl.Length() > 0 {
				lang = detectCodeLang(codeEl)
				codeText = codeEl.Text()
			} else {
				codeText = s.Text()
			}
			buf.WriteString("\n\n```" + lang + "\n")
			buf.WriteString(codeText)
			if !strings.HasSuffix(codeText, "\n") {
				buf.WriteString("\n")
			}
			buf.WriteString("```\n\n")

		case "a":
			href, exists := s.Attr("href")
			text := strings.TrimSpace(s.Text())
			if text == "" {
				text = href
			}
			if exists && href != "" {
				buf.WriteString("[" + text + "](" + href + ")")
			} else {
				buf.WriteString(text)
			}

		case "img":
			src, _ := s.Attr("src")
			alt, _ := s.Attr("alt")
			if src != "" {
				buf.WriteString("![" + alt + "](" + src + ")")
			} else if alt != "" {
				buf.WriteString(alt)
			}

		case "ul":
			buf.WriteString("\n")
			s.Children().Each(func(j int, li *goquery.Selection) {
				if goquery.NodeName(li) == "li" {
					indent := strings.Repeat("  ", depth)
					buf.WriteString(indent + "- ")
					convertLI(buf, li, depth)
					buf.WriteString("\n")
				}
			})
			buf.WriteString("\n")

		case "ol":
			buf.WriteString("\n")
			s.Children().Each(func(j int, li *goquery.Selection) {
				if goquery.NodeName(li) == "li" {
					indent := strings.Repeat("  ", depth)
					buf.WriteString(fmt.Sprintf("%s%d. ", indent, j+1))
					convertLI(buf, li, depth)
					buf.WriteString("\n")
				}
			})
			buf.WriteString("\n")

		case "blockquote":
			var inner strings.Builder
			convertNodes(&inner, s, depth, true)
			lines := strings.Split(strings.TrimSpace(inner.String()), "\n")
			buf.WriteString("\n\n")
			for _, line := range lines {
				buf.WriteString("> " + line + "\n")
			}
			buf.WriteString("\n")

		case "table":
			convertTable(buf, s)

		case "dl":
			convertDefList(buf, s)

		case "div", "section", "article", "main", "span", "figure", "figcaption":
			convertNodes(buf, s, depth, inBlockquote)

		default:
			convertNodes(buf, s, depth, inBlockquote)
		}
	})
}

func convertLI(buf *strings.Builder, li *goquery.Selection, depth int) {
	// Check for nested lists
	hasNestedList := li.Find("ul, ol").Length() > 0

	if hasNestedList {
		// Render direct text content first
		li.Contents().Each(func(i int, s *goquery.Selection) {
			node := s.Get(0)
			if node.Type == html.TextNode {
				text := strings.TrimSpace(node.Data)
				if text != "" {
					buf.WriteString(text)
				}
			} else if node.Type == html.ElementNode {
				tag := goquery.NodeName(s)
				if tag == "ul" || tag == "ol" {
					buf.WriteString("\n")
					convertNodes(buf, s, depth+1, false)
				} else {
					convertNodes(buf, s, depth, false)
				}
			}
		})
	} else {
		convertNodes(buf, li, depth, false)
	}
}

func convertTable(buf *strings.Builder, table *goquery.Selection) {
	var headers []string
	var rows [][]string

	table.Find("thead tr").First().Find("th").Each(func(i int, th *goquery.Selection) {
		headers = append(headers, strings.TrimSpace(th.Text()))
	})

	// If no thead, try first row as header
	if len(headers) == 0 {
		table.Find("tr").First().Find("th, td").Each(func(i int, cell *goquery.Selection) {
			headers = append(headers, strings.TrimSpace(cell.Text()))
		})
	}

	table.Find("tbody tr").Each(func(i int, tr *goquery.Selection) {
		var row []string
		tr.Find("td").Each(func(j int, td *goquery.Selection) {
			row = append(row, strings.TrimSpace(td.Text()))
		})
		if len(row) > 0 {
			rows = append(rows, row)
		}
	})

	if len(headers) == 0 {
		return
	}

	buf.WriteString("\n\n")
	buf.WriteString("| " + strings.Join(headers, " | ") + " |\n")
	seps := make([]string, len(headers))
	for i := range seps {
		seps[i] = "---"
	}
	buf.WriteString("| " + strings.Join(seps, " | ") + " |\n")
	for _, row := range rows {
		// Pad row to header length
		for len(row) < len(headers) {
			row = append(row, "")
		}
		buf.WriteString("| " + strings.Join(row[:len(headers)], " | ") + " |\n")
	}
	buf.WriteString("\n")
}

func convertDefList(buf *strings.Builder, dl *goquery.Selection) {
	buf.WriteString("\n\n")
	dl.Children().Each(func(i int, s *goquery.Selection) {
		tag := goquery.NodeName(s)
		switch tag {
		case "dt":
			buf.WriteString("**" + strings.TrimSpace(s.Text()) + "**\n")
		case "dd":
			buf.WriteString(": " + strings.TrimSpace(s.Text()) + "\n\n")
		}
	})
}

func detectCodeLang(code *goquery.Selection) string {
	class, _ := code.Attr("class")
	if class == "" {
		return ""
	}
	// Common patterns: language-go, lang-go, highlight-go
	for _, cls := range strings.Fields(class) {
		for _, prefix := range []string{"language-", "lang-", "highlight-"} {
			if strings.HasPrefix(cls, prefix) {
				return strings.TrimPrefix(cls, prefix)
			}
		}
	}
	return ""
}

func collapseWhitespace(s string) string {
	// Replace sequences of whitespace with single space
	var buf strings.Builder
	inSpace := false
	for _, r := range s {
		if r == ' ' || r == '\t' || r == '\n' || r == '\r' {
			if !inSpace {
				buf.WriteRune(' ')
				inSpace = true
			}
		} else {
			buf.WriteRune(r)
			inSpace = false
		}
	}
	return buf.String()
}

func cleanMarkdown(s string) string {
	// Collapse 3+ newlines to 2
	for strings.Contains(s, "\n\n\n") {
		s = strings.ReplaceAll(s, "\n\n\n", "\n\n")
	}
	return strings.TrimSpace(s)
}

package main

import (
	"regexp"
	"strings"

	"github.com/PuerkitoBio/goquery"
)

// Tags to remove entirely (tag + all contents).
var killTags = []string{
	"script", "style", "noscript", "iframe", "svg", "canvas",
	"video", "audio", "source", "embed", "object",
	"form", "input", "button", "select", "textarea",
	"option", "label", "fieldset",
}

// Landmark tags to remove.
var landmarkTags = []string{"nav", "footer", "aside"}

// Class/ID patterns indicating noise.
var noisePattern = regexp.MustCompile(
	`(?i)(cookie|consent|gdpr|banner|popup|modal|overlay|` +
		`newsletter|subscribe|social|share|sidebar|` +
		`advertisement|ad-|promo|related-posts)`,
)

// stripNoise removes non-content elements from the document.
func stripNoise(doc *goquery.Document) {
	// 1. Kill tags entirely
	for _, tag := range killTags {
		doc.Find(tag).Remove()
	}

	// 2. Kill landmark tags
	for _, tag := range landmarkTags {
		doc.Find(tag).Remove()
	}

	// 3. Kill elements with noisy class/id
	doc.Find("*").Each(func(i int, s *goquery.Selection) {
		class, _ := s.Attr("class")
		id, _ := s.Attr("id")
		combined := class + " " + id
		if noisePattern.MatchString(combined) {
			s.Remove()
		}
	})

	// 4. Kill header elements that don't contain an h1
	doc.Find("header").Each(func(i int, s *goquery.Selection) {
		if s.Find("h1").Length() == 0 {
			s.Remove()
		}
	})

	// 5. Kill hidden elements
	doc.Find("[style]").Each(func(i int, s *goquery.Selection) {
		style, _ := s.Attr("style")
		if strings.Contains(style, "display:none") || strings.Contains(style, "display: none") {
			s.Remove()
		}
	})
	doc.Find(`[aria-hidden="true"]`).Remove()

	// 6. Remove HTML comments (goquery doesn't expose them easily, but they're in the DOM)
	doc.Find("*").Contents().Each(func(i int, s *goquery.Selection) {
		if s.Get(0) != nil && s.Get(0).Type == 8 { // CommentNode
			s.Remove()
		}
	})
}

// extractContent picks the main content container from a cleaned document.
func extractContent(doc *goquery.Document, domain string) *goquery.Selection {
	// 1. Try site-specific selector
	if sel, ok := siteSelector(domain); ok {
		found := doc.Find(sel)
		if found.Length() > 0 {
			return found.First()
		}
	}

	// 2. Try semantic tags
	for _, selector := range []string{"main", "article", `[role="main"]`} {
		found := doc.Find(selector)
		if found.Length() > 0 {
			return found.First()
		}
	}

	// 3. Readability scoring fallback
	best := doc.Find("body")
	bestScore := 0

	doc.Find("div, section").Each(func(i int, s *goquery.Selection) {
		score := readabilityScore(s)
		if score > bestScore {
			bestScore = score
			best = s
		}
	})

	return best
}

func readabilityScore(s *goquery.Selection) int {
	text := s.Text()
	score := len(strings.TrimSpace(text))

	// Bonus for <p> and <pre> children
	score += s.Find("p").Length() * 50
	score += s.Find("pre").Length() * 100

	// Penalty for high link density
	linkText := 0
	s.Find("a").Each(func(i int, a *goquery.Selection) {
		linkText += len(a.Text())
	})
	totalText := len(text)
	if totalText > 0 {
		linkRatio := float64(linkText) / float64(totalText)
		if linkRatio > 0.5 {
			score = score / 3
		}
	}

	// Penalty for noise patterns in class/id
	class, _ := s.Attr("class")
	id, _ := s.Attr("id")
	if noisePattern.MatchString(class + " " + id) {
		score = 0
	}

	return score
}

package main

import (
	"fmt"
	"io"
	"net/http"
	"net/http/cookiejar"
	"strings"
	"time"
)

type fetchOptions struct {
	timeout    time.Duration
	maxRetries int
	verbose    bool
	quiet      bool
	noTLSSpoof bool
	noHeadless bool
	headers    []string
}

type fetchResult struct {
	html string
	tier int
	url  string
}

var browserHeaders = map[string]string{
	"User-Agent":      "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/125.0.0.0 Safari/537.36",
	"Accept":          "text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,*/*;q=0.8",
	"Accept-Language": "en-US,en;q=0.5",
	"Accept-Encoding": "gzip, deflate, br",
	"DNT":             "1",
	"Connection":      "keep-alive",
	"Upgrade-Insecure-Requests": "1",
	"Sec-Fetch-Dest":  "document",
	"Sec-Fetch-Mode":  "navigate",
	"Sec-Fetch-Site":  "none",
	"Sec-Fetch-User":  "?1",
}

func logVerbose(opts *fetchOptions, format string, args ...any) {
	if opts.verbose && !opts.quiet {
		fmt.Fprintf(stderr, format+"\n", args...)
	}
}

// fetch retrieves the HTML of a URL, escalating through tiers as needed.
func fetch(rawURL string, opts *fetchOptions) (*fetchResult, error) {
	// Tier 1: plain HTTP
	result, err := fetchTier1(rawURL, opts)
	if err == nil {
		return result, nil
	}

	// Check if we should escalate
	if isEscalatable(err) {
		// Tier 2: utls TLS spoofing
		if !opts.noTLSSpoof {
			logVerbose(opts, "[tier1] %s → escalating to tier2", rawURL)
			result, err = fetchTier2(rawURL, opts)
			if err == nil {
				return result, nil
			}
		}

		// Tier 3: headless Chrome
		if !opts.noHeadless {
			logVerbose(opts, "[tier2] %s → escalating to tier3", rawURL)
			result, err = fetchTier3(rawURL, opts)
			if err == nil {
				return result, nil
			}
		}
	}

	return nil, err
}

type httpError struct {
	statusCode int
	message    string
}

func (e *httpError) Error() string {
	return fmt.Sprintf("HTTP %d: %s", e.statusCode, e.message)
}

func isEscalatable(err error) bool {
	if he, ok := err.(*httpError); ok {
		return he.statusCode == 403 || he.statusCode == 503
	}
	return false
}

func fetchTier1(rawURL string, opts *fetchOptions) (*fetchResult, error) {
	jar, _ := cookiejar.New(nil)
	client := &http.Client{
		Timeout: opts.timeout,
		Jar:     jar,
	}

	var lastErr error
	for attempt := 0; attempt < opts.maxRetries; attempt++ {
		if attempt > 0 {
			backoff := time.Duration(1<<uint(attempt-1)) * time.Second
			if backoff > 30*time.Second {
				backoff = 30 * time.Second
			}
			time.Sleep(backoff)
		}

		req, err := http.NewRequest("GET", rawURL, nil)
		if err != nil {
			return nil, err
		}

		for k, v := range browserHeaders {
			req.Header.Set(k, v)
		}
		for _, h := range opts.headers {
			parts := strings.SplitN(h, "=", 2)
			if len(parts) == 2 {
				req.Header.Set(parts[0], parts[1])
			}
		}

		resp, err := client.Do(req)
		if err != nil {
			lastErr = err
			logVerbose(opts, "[tier1] %s → error: %v (attempt %d/%d)", rawURL, err, attempt+1, opts.maxRetries)
			continue
		}

		body, err := io.ReadAll(resp.Body)
		resp.Body.Close()
		if err != nil {
			lastErr = err
			continue
		}

		logVerbose(opts, "[tier1] %s → %d", rawURL, resp.StatusCode)

		if resp.StatusCode == 200 {
			return &fetchResult{
				html: string(body),
				tier: 1,
				url:  rawURL,
			}, nil
		}

		if resp.StatusCode == 429 {
			// Rate limited — retry with backoff
			lastErr = &httpError{statusCode: 429, message: "rate limited"}
			continue
		}

		if resp.StatusCode >= 500 {
			lastErr = &httpError{statusCode: resp.StatusCode, message: "server error"}
			continue
		}

		// 403, 503 — don't retry, escalate
		if resp.StatusCode == 403 || resp.StatusCode == 503 {
			return nil, &httpError{statusCode: resp.StatusCode, message: "access denied"}
		}

		// Other client errors — don't retry
		return nil, &httpError{statusCode: resp.StatusCode, message: http.StatusText(resp.StatusCode)}
	}

	return nil, fmt.Errorf("fetch failed after %d retries: %w", opts.maxRetries, lastErr)
}

package main

import (
	"fmt"
	"net/url"
	"os"
	"path"
	"path/filepath"
	"sort"
	"strings"
	"sync"
	"time"

	"github.com/PuerkitoBio/goquery"
)

type urlSet struct {
	mu   sync.Mutex
	urls map[string]bool
}

func newURLSet() *urlSet {
	return &urlSet{urls: make(map[string]bool)}
}

func (s *urlSet) add(u string) bool {
	s.mu.Lock()
	defer s.mu.Unlock()
	normalized := normalizeURL(u)
	if s.urls[normalized] {
		return false
	}
	s.urls[normalized] = true
	return true
}

func normalizeURL(rawURL string) string {
	u, err := url.Parse(rawURL)
	if err != nil {
		return rawURL
	}
	// Strip fragment
	u.Fragment = ""
	// Strip trailing slash
	u.Path = strings.TrimRight(u.Path, "/")
	// Sort query params
	if u.RawQuery != "" {
		params := u.Query()
		keys := make([]string, 0, len(params))
		for k := range params {
			keys = append(keys, k)
		}
		sort.Strings(keys)
		var parts []string
		for _, k := range keys {
			for _, v := range params[k] {
				parts = append(parts, url.QueryEscape(k)+"="+url.QueryEscape(v))
			}
		}
		u.RawQuery = strings.Join(parts, "&")
	}
	return u.String()
}

func isSameDomain(href string, base *url.URL) bool {
	if strings.HasPrefix(href, "/") {
		return true
	}
	u, err := url.Parse(href)
	if err != nil {
		return false
	}
	// Strip www prefix for comparison
	baseHost := strings.TrimPrefix(base.Hostname(), "www.")
	hrefHost := strings.TrimPrefix(u.Hostname(), "www.")
	return baseHost == hrefHost
}

func pathMatchesGlob(urlPath, pattern string) bool {
	matched, _ := path.Match(pattern, urlPath)
	if matched {
		return true
	}
	// Also try matching the last path segment
	base := path.Base(urlPath)
	matched, _ = path.Match(pattern, "/"+base)
	return matched
}

func extractLinks(doc *goquery.Document, base *url.URL) []string {
	var links []string
	doc.Find("a[href]").Each(func(i int, s *goquery.Selection) {
		href, _ := s.Attr("href")
		if href == "" {
			return
		}
		// Skip javascript:, mailto:, tel:, etc.
		if strings.HasPrefix(href, "javascript:") || strings.HasPrefix(href, "mailto:") || strings.HasPrefix(href, "tel:") {
			return
		}
		// Skip fragment-only links
		if strings.HasPrefix(href, "#") {
			return
		}

		// Resolve relative URLs
		resolved, err := base.Parse(href)
		if err != nil {
			return
		}

		// Must be same domain
		if !isSameDomain(resolved.String(), base) {
			return
		}

		// Must be http/https
		if resolved.Scheme != "http" && resolved.Scheme != "https" {
			return
		}

		links = append(links, resolved.String())
	})
	return links
}

type crawlResult struct {
	url      string
	title    string
	markdown string
	err      error
}

func runCrawl(cfg *config, seedURL *url.URL) error {
	seen := newURLSet()
	seen.add(seedURL.String())

	type queueItem struct {
		url   string
		depth int
	}

	queue := []queueItem{{url: seedURL.String(), depth: 0}}
	var results []crawlResult
	var mu sync.Mutex
	pageCount := 0
	domainFailures := 0
	maxDomainFailures := 5

	sem := make(chan struct{}, cfg.concurrency)

	// Set up max-time deadline
	deadline := time.Now().Add(cfg.maxTime)

	opts := &fetchOptions{
		timeout:    cfg.timeout,
		maxRetries: cfg.maxRetries,
		verbose:    cfg.verbose,
		quiet:      cfg.quiet,
		noTLSSpoof: cfg.noTLSSpoof,
		noHeadless: cfg.noHeadless,
		headers:    cfg.headers,
	}

	for len(queue) > 0 {
		if time.Now().After(deadline) {
			if !cfg.quiet {
				fmt.Fprintf(stderr, "[crawl] max-time reached, stopping\n")
			}
			break
		}

		if pageCount >= cfg.maxPages {
			break
		}

		if domainFailures >= maxDomainFailures {
			if !cfg.quiet {
				fmt.Fprintf(stderr, "[crawl] too many domain failures (%d), stopping\n", domainFailures)
			}
			break
		}

		// Dequeue batch up to concurrency
		batchSize := cfg.concurrency
		if batchSize > len(queue) {
			batchSize = len(queue)
		}
		if batchSize > cfg.maxPages-pageCount {
			batchSize = cfg.maxPages - pageCount
		}
		batch := queue[:batchSize]
		queue = queue[batchSize:]

		var wg sync.WaitGroup
		for _, item := range batch {
			wg.Add(1)
			go func(qi queueItem) {
				defer wg.Done()
				sem <- struct{}{}
				defer func() { <-sem }()

				// Rate limiting delay
				if cfg.delay > 0 {
					time.Sleep(cfg.delay)
				}

				result, err := fetch(qi.url, opts)
				if err != nil {
					mu.Lock()
					domainFailures++
					results = append(results, crawlResult{url: qi.url, err: err})
					if cfg.verbose {
						fmt.Fprintf(stderr, "[crawl] %s → error: %v\n", qi.url, err)
					}
					mu.Unlock()
					return
				}

				doc, err := goquery.NewDocumentFromReader(strings.NewReader(result.html))
				if err != nil {
					mu.Lock()
					results = append(results, crawlResult{url: qi.url, err: err})
					mu.Unlock()
					return
				}

				stripNoise(doc)
				content := extractContent(doc, seedURL.Host)
				markdown := convertToMarkdown(content)
				title := doc.Find("title").First().Text()

				mu.Lock()
				pageCount++
				num := pageCount
				results = append(results, crawlResult{url: qi.url, title: title, markdown: markdown})
				if cfg.verbose {
					fmt.Fprintf(stderr, "[tier%d] %s → 200 [%d/%d]\n", result.tier, qi.url, num, cfg.maxPages)
				}
				mu.Unlock()

				// Discover new links if within depth
				if qi.depth < cfg.depth {
					links := extractLinks(doc, seedURL)
					mu.Lock()
					for _, link := range links {
						normalized := normalizeURL(link)
						if !seen.add(normalized) {
							continue
						}

						linkURL, err := url.Parse(normalized)
						if err != nil {
							continue
						}

						// Apply include/exclude filters
						if cfg.include != "" && !pathMatchesGlob(linkURL.Path, cfg.include) {
							continue
						}
						if cfg.exclude != "" && pathMatchesGlob(linkURL.Path, cfg.exclude) {
							continue
						}

						queue = append(queue, queueItem{url: normalized, depth: qi.depth + 1})
					}
					mu.Unlock()
				}
			}(item)
		}
		wg.Wait()
	}

	// Write output
	if cfg.output != "" {
		return writeCrawlOutput(cfg, results)
	}

	// stdout: concatenate all pages
	for _, r := range results {
		if r.err != nil {
			continue
		}
		fmt.Fprintf(os.Stdout, "---\n# %s\n> %s\n\n%s\n\n", r.title, r.url, r.markdown)
	}

	// Summary
	failed := 0
	for _, r := range results {
		if r.err != nil {
			failed++
		}
	}
	if !cfg.quiet {
		fmt.Fprintf(stderr, "[done] %d/%d pages fetched, %d failed\n", len(results)-failed, cfg.maxPages, failed)
	}

	return nil
}

func writeCrawlOutput(cfg *config, results []crawlResult) error {
	if err := os.MkdirAll(cfg.output, 0755); err != nil {
		return fmt.Errorf("create output directory: %w", err)
	}

	// Write individual page files
	var indexEntries []string
	for _, r := range results {
		if r.err != nil {
			continue
		}

		filename := urlToFilename(r.url) + ".md"
		filePath := filepath.Join(cfg.output, filename)

		if err := os.WriteFile(filePath, []byte(r.markdown), 0644); err != nil {
			return fmt.Errorf("write %s: %w", filePath, err)
		}

		title := r.title
		if title == "" {
			title = r.url
		}
		indexEntries = append(indexEntries, fmt.Sprintf("- [%s](%s) — %s", title, filename, r.url))
	}

	// Write index.md
	var index strings.Builder
	index.WriteString("# Crawl Index\n\n")
	index.WriteString(fmt.Sprintf("Pages: %d\n\n", len(indexEntries)))
	for _, entry := range indexEntries {
		index.WriteString(entry + "\n")
	}

	indexPath := filepath.Join(cfg.output, "index.md")
	if err := os.WriteFile(indexPath, []byte(index.String()), 0644); err != nil {
		return fmt.Errorf("write index: %w", err)
	}

	failed := 0
	for _, r := range results {
		if r.err != nil {
			failed++
		}
	}
	if !cfg.quiet {
		fmt.Fprintf(stderr, "[done] %d pages written to %s, %d failed\n", len(indexEntries), cfg.output, failed)
	}

	return nil
}

func urlToFilename(rawURL string) string {
	u, err := url.Parse(rawURL)
	if err != nil {
		return "page"
	}
	p := strings.TrimPrefix(u.Path, "/")
	p = strings.TrimSuffix(p, "/")
	if p == "" {
		return "index"
	}
	// Replace slashes with dashes
	p = strings.ReplaceAll(p, "/", "-")
	// Clean up any weird chars
	p = strings.Map(func(r rune) rune {
		if (r >= 'a' && r <= 'z') || (r >= 'A' && r <= 'Z') || (r >= '0' && r <= '9') || r == '-' || r == '_' {
			return r
		}
		return '-'
	}, p)
	return p
}
